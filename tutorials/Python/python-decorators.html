Python Decorators

Decorators are functions that wrap other functions to extend or modify their behavior.

They are often used for logging, timing, authentication, and caching.

Basic decorator

def logger(func):
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__}")
        result = func(*args, **kwargs)
        print(f"{func.__name__} returned {result}")
        return result
    return wrapper

@logger
def add(a, b):
    return a + b

add(2, 3)

The @logger syntax is shorthand for add = logger(add).

Preserving function metadata with wraps

import functools

def timer(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        import time
        start = time.time()
        result = func(*args, **kwargs)
        duration = time.time() - start
        print(f"{func.__name__} took {duration:.4f} seconds")
        return result
    return wrapper

@timer
def slow():
    sum(range(1_000_000))

slow()
